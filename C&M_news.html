const PRIMARY_KEYWORDS = "통신 미디어 우주 IoT 6G 표준 국제 표준 표준화 AI 코덱 MPEG 3GPP ITU 반도체"; const SECONDARY_KEYWORDS = "5G 네트워크 위성 AI 반도체 데이터센터 XR 플랫폼"; function fetchNews(keywords) { const rss = "https://news.google.com/rss/search?q=" + encodeURIComponent(keywords) + "&hl=ko&gl=KR&ceid=KR:ko"; const api = "https://api.rss2json.com/v1/api.json?rss_url=" + encodeURIComponent(rss); return fetch(api).then(res => res.json()); } function processItems(items) { const now = new Date(); const oneMonthAgo = new Date(); oneMonthAgo.setMonth(now.getMonth() - 1); return items.filter(item => { const pub = new Date(item.pubDate); return pub >= oneMonthAgo; }); } function render(items) { const container = document.getElementById("news"); container.innerHTML = ""; items.slice(0, 10).forEach(item => { const title = item.title; const link = item.link; const date = new Date(item.pubDate).toLocaleDateString("ko-KR"); let category = "기타"; if (title.includes("6G")) category = "6G"; else if (title.includes("우주") || title.includes("위성")) category = "우주"; else if (title.includes("IoT")) category = "IoT"; else if (title.includes("표준") || title.includes("3GPP") || title.includes("ITU")) category = "표준"; else if (title.includes("AI")) category = "AI"; else if (title.includes("코덱") || title.includes("MPEG")) category = "코덱"; else if (title.includes("통신")) category = "통신"; let signalClass = "green"; if (title.includes("투자") || title.includes("출시") || title.includes("확장")) signalClass = "red"; else if (title.includes("협력") || title.includes("개발")) signalClass = "yellow"; container.innerHTML += `
📅 ${date}
${title}
${category}
`; }); } // ✅ 실행 로직 fetchNews(PRIMARY_KEYWORDS).then(data1 => { let primaryItems = processItems(data1.items); if (primaryItems.length >= 7) { render(primaryItems); } else { // ✅ 2차 검색 (fallback) fetchNews(SECONDARY_KEYWORDS).then(data2 => { let secondaryItems = processItems(data2.items); // ✅ 중복 제거 (링크 기준) const combined = [...primaryItems]; secondaryItems.forEach(item => { if (!combined.some(i => i.link === item.link)) { combined.push(item); } }); render(combined); }); } }).catch(() => { document.getElementById("news").innerHTML = "뉴스 로드 실패"; });
